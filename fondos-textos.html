🖼️ Fondos y Textos

Personaliza el aspecto visual y agrega textos informativos

🎨 Cambiar el Fondo

Puedes personalizar el fondo del tablero usando game.ground():

import wollok.game.*

program juegoConFondo {
    game.width(10)
    game.height(7)
    game.title("Juego con Fondo Personalizado")
    
    // Cambiar el fondo del tablero
    game.ground("cesped.png")
    
    game.start()
}

🖼️ Tipos de Fondos Comunes

  • Cesped: Para juegos de aventura al aire libre
  • Piedra: Para mazmorras y castillos
  • Agua: Para niveles acuáticos
  • Nieve: Para niveles invernales
  • Arena: Para desiertos
  • Madera: Para interiores

💬 Mostrar Textos

Puedes mostrar mensajes de texto en pantalla usando game.say():

import wollok.game.*

program juegoConTextos {
    const heroe = new Heroe()
    game.addVisualCharacter(heroe)
    
    // Mensaje simple
    game.say("¡Bienvenido al juego!")
    
    // Mensaje con información del objeto
    game.schedule(2000, {
        game.say("Posición del héroe: " + heroe.position)
    })
    
    // Mensaje con variables
    var puntos = 0
    game.schedule(5000, {
        puntos = puntos + 10
        game.say("Puntos: " + puntos)
    })
    
    game.start()
}

🎨 Colorear Textos

Puedes cambiar el color de los textos usando game.say() con parámetros de color:

import wollok.game.*

program textosColoridos {
    const heroe = new Heroe()
    game.addVisualCharacter(heroe)
    
    // Texto en diferentes colores
    game.say("¡Texto rojo!", "red")
    game.say("¡Texto azul!", "blue")
    game.say("¡Texto verde!", "green")
    game.say("¡Texto amarillo!", "yellow")
    game.say("¡Texto blanco!", "white")
    game.say("¡Texto negro!", "black")
    
    game.start()
}

🌈 Colores Disponibles

  • Básicos: red, blue, green, yellow, white, black
  • Otros: orange, purple, pink, cyan, magenta
  • Sin color: Si no especificas color, usa el color por defecto

🖼️ Imágenes con Texto

Puedes combinar imágenes con texto para crear mensajes más visuales:

import wollok.game.*

program imagenesConTexto {
    const heroe = new Heroe()
    game.addVisualCharacter(heroe)
    
    // Mostrar imagen con texto
    game.say("¡Power-up!", "powerup.png")
    game.say("¡Vida extra!", "vida.png")
    game.say("¡Moneda!", "moneda.png")
    
    // Texto con imagen y color
    game.say("¡Nivel completado!", "estrella.png", "gold")
    
    game.start()
}

🎨 Combinaciones Visuales

  • Iconos de estado: Corazones para vida, estrellas para puntos
  • Indicadores de acción: Flechas para movimiento, espadas para ataque
  • Elementos del juego: Monedas, gemas, power-ups

📊 Sistema de Información en Pantalla

Puedes crear un sistema completo de información que se actualice constantemente:

import wollok.game.*

class Heroe {
    var position = game.at(5, 5)
    var vida = 100
    var puntos = 0
    var nivel = 1
    method visual() = "hero.png"
    
    method mostrarInfo() {
        game.say("Nivel: " + nivel + " | Vida: " + vida + " | Puntos: " + puntos)
    }
    
    method ganarPuntos(cantidad) {
        puntos = puntos + cantidad
        game.say("¡+" + cantidad + " puntos!", "moneda.png", "gold")
    }
    
    method recibirDano(cantidad) {
        vida = vida - cantidad
        game.say("¡-" + cantidad + " vida!", "corazon.png", "red")
        if (vida <= 0) {
            game.say("¡Game Over!", "skull.png", "red")
        }
    }
}

program juegoConInfo {
    const heroe = new Heroe()
    game.addVisualCharacter(heroe)
    
    // Mostrar información cada 3 segundos
    game.schedule(3000, { heroe.mostrarInfo() })
    
    // Controles para probar
    keyboard.p().onPressDo({ heroe.ganarPuntos(10) })
    keyboard.d().onPressDo({ heroe.recibirDano(20) })
    keyboard.i().onPressDo({ heroe.mostrarInfo() })
    
    game.start()
}

🎭 Diálogos y Conversaciones

Puedes crear sistemas de diálogo más complejos:

import wollok.game.*

class NPC {
    var position = game.at(8, 3)
    var nombre = "Merchant"
    method visual() = "npc.png"
    
    method hablar() {
        game.say("¡Hola, aventurero!", "npc.png", "blue")
        game.schedule(2000, {
            game.say("¿Necesitas algo?", "npc.png", "blue")
        })
        game.schedule(4000, {
            game.say("Tengo objetos mágicos...", "npc.png", "blue")
        })
    }
}

program juegoConDialogos {
    const heroe = new Heroe()
    const npc = new NPC()
    
    game.addVisualCharacter(heroe)
    game.addVisual(npc)
    
    // Hablar con NPC
    keyboard.h().onPressDo({ npc.hablar() })
    
    game.start()
}

🎨 Personalización Avanzada

Puedes crear fondos dinámicos que cambien según el estado del juego:

import wollok.game.*

var nivelActual = 1
var esNoche = false

method cambiarFondo() {
    if (esNoche) {
        game.ground("noche.png")
        game.say("¡Se hizo de noche!", "luna.png", "white")
    } else {
        game.ground("dia.png")
        game.say("¡Amanece!", "sol.png", "yellow")
    }
    esNoche = !esNoche
}

method cambiarNivel() {
    nivelActual = nivelActual + 1
    if (nivelActual == 2) {
        game.ground("bosque.png")
        game.say("¡Entraste al bosque!", "arbol.png", "green")
    } else if (nivelActual == 3) {
        game.ground("cueva.png")
        game.say("¡Descendiste a la cueva!", "piedra.png", "gray")
    }
}

program juegoPersonalizado {
    const heroe = new Heroe()
    game.addVisualCharacter(heroe)
    
    // Cambiar entre día y noche cada 10 segundos
    game.schedule(10000, { cambiarFondo() })
    
    // Cambiar nivel cada 30 segundos
    game.schedule(30000, { cambiarNivel() })
    
    // Controles
    keyboard.f().onPressDo({ cambiarFondo() })
    keyboard.n().onPressDo({ cambiarNivel() })
    
    game.start()
}

📱 Interfaz de Usuario

Puedes crear una interfaz de usuario completa con información constante:

import wollok.game.*

class UI {
    method mostrarHUD(heroe) {
        game.say("=== HUD ===")
        game.say("Vida: " + heroe.vida + "/100", "corazon.png", "red")
        game.say("Puntos: " + heroe.puntos, "moneda.png", "gold")
        game.say("Nivel: " + heroe.nivel, "estrella.png", "yellow")
        game.say("===========")
    }
    
    method mostrarMenu() {
        game.say("=== MENU ===")
        game.say("H - Mostrar ayuda")
        game.say("I - Mostrar información")
        game.say("P - Pausar juego")
        game.say("============")
    }
}

program juegoConUI {
    const heroe = new Heroe()
    const ui = new UI()
    
    game.addVisualCharacter(heroe)
    
    // Mostrar HUD cada 5 segundos
    game.schedule(5000, { ui.mostrarHUD(heroe) })
    
    // Controles de interfaz
    keyboard.h().onPressDo({ ui.mostrarMenu() })
    keyboard.i().onPressDo({ ui.mostrarHUD(heroe) })
    
    game.start()
}

💡 Consejos Importantes

  • Usa colores consistentes para diferentes tipos de mensajes
  • No satures la pantalla con demasiados textos
  • Usa imágenes pequeñas para los iconos
  • Planifica la interfaz antes de implementarla
  • Prueba en diferentes tamaños de pantalla
